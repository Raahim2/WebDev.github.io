// Function to create the grid function createGrid() { const container = document.getElementById('gridContainer'); for (let i = 0; i < 100; i++) { for (let j = 0; j < 25; j++) { const cell = document.createElement('div'); cell.classList.add('cell'); // Assigning classes based on probabilities const randomNumber = Math.random() * 100; // Generate a number between 0 and 100 if (randomNumber < 20) { cell.classList.add('im1'); } else if (randomNumber < 40) { cell.classList.add('im2'); } else if (randomNumber < 60) { cell.classList.add('im3'); } else if (randomNumber < 70) { cell.classList.add('im4'); } else if (randomNumber < 80) { cell.classList.add('im5'); } else if (randomNumber < 90) { cell.classList.add('im6'); } else if (randomNumber < 95) { cell.classList.add('im7'); } else if (randomNumber < 97) { cell.classList.add('im8'); } else if (randomNumber < 99) { cell.classList.add('im9'); } else { cell.classList.add('im10'); } container.appendChild(cell); } } }
